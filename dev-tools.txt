# Dev tools for aozora-proof — consumed by `just setup` (via cargo-binstall).
#
# Format: one `crate[@version-req]` per line. `@0.9` is a semver requirement
# (^0.9), so patch updates flow in automatically while the minor stays pinned
# for local == CI parity. Blank lines and `#` comments are ignored.
#
# Keep these minors in step with the install-action versions in CI
# (.github/actions/setup-rust and the `lint` job in .github/workflows/ci.yml).

cargo-nextest@0.9      # test runner — CI's `test` job uses this
cargo-deny@0.19        # supply-chain / license gate — CI's `deny` job
cargo-llvm-cov@0.8     # coverage — CI's `coverage` job
wasm-pack@0.15         # builds the browser WASM package (web/ dev loop)
typos-cli@1.47         # spell check — CI's `lint` job
bacon@3                # background watch loop (see bacon.toml)
watchexec-cli@2.5      # rebuilds wasm on Rust changes for `just web`
taplo-cli@0.10         # TOML format + lint — CI's `lint` job

# Non-cargo tools: shellcheck via taiki-e/install-action and actionlint via a
# pinned release download (CI's `lint` job) — install locally via mise or your
# package manager. The web app (`web/`) is a SvelteKit project: install Node 24+
# and pnpm, then `pnpm -C web install` (or run `just lint-web`).
